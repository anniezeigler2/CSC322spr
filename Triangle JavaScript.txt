const canvas = document.querySelector('canvas');
const ctx = canvas.getContext('2d');

// Set line style (color, width)
ctx.strokeStyle = 'black'; 
ctx.lineWidth = 2;



// Draw a line from (10, 10) to (100, 50)
ctx.beginPath();

ctx.moveTo(200, 100);
ctx.lineTo(100, 300);
ctx.lineTo(300,300);
ctx.lineTo(200,100);

ctx.stroke();
